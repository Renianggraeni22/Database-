-- pembuatan table jenis_produk
CREATE TABLE jenis_produk (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    nama VARCHAR(45) NOT NULL
) ENGINE=InnoDB;

-- pembuatan table produk
CREATE TABLE produk (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    kode VARCHAR(10) NOT NULL UNIQUE,
    nama VARCHAR(45) NOT NULL,
    harga_beli DOUBLE,
    harga_jual DOUBLE,
    stok INT,
    min_stok INT,
    jenis_produk_id INT,
    FOREIGN KEY (jenis_produk_id) REFERENCES jenis_produk(id)
);

-- pembuatan table vendor
CREATE TABLE vendor (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    kode VARCHAR(10) NOT NULL UNIQUE,
    nama VARCHAR(45) NOT NULL,
    alamat VARCHAR(100),
    telepon VARCHAR(20)
);

-- pembuatan table pembelian
CREATE TABLE pembelian (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    tanggal DATE,
    total DOUBLE,
    vendor_id INT,
    FOREIGN KEY (vendor_id) REFERENCES vendor(id)
);

-- pembuatan table pesanan_items
CREATE TABLE pesanan_items (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    pesanan_id INT NOT NULL,
    produk_id INT NOT NULL,
    jumlah INT NOT NULL,
    harga DOUBLE NOT NULL,
    FOREIGN KEY (pesanan_id) REFERENCES pesanan(id),
    FOREIGN KEY (produk_id) REFERENCES produk(id)
) ENGINE=InnoDB;

-- menampilkan detail pesanan dengan join pelanggan & produk
SELECT p.id AS no_pesanan, p.tanggal, pl.nama AS pelanggan, pr.nama AS produk,
       pi.jumlah, pi.harga
FROM pesanan p
JOIN pelanggan pl ON p.pelanggan_id = pl.id
JOIN pesanan_items pi ON p.id = pi.pesanan_id
JOIN produk pr ON pi.produk_id = pr.id;
